[总结]数据结构进阶

0.基础位运算/操作技巧:

1<<n=2^n n<<1=2*n;

状态压缩的相关操作：

取出整数n在二进制表示下的第k位：(n>>k)&1;

取出整数n在二进制表示下的0~k-1位：n&((1<<k)-1);

把整数n在二进制表示下的第k位取反：n^(1<<k);

对整数n在二进制表示下的第k位赋值1：n|(1<<k);

对整数n在二进制表示下的第k位赋值0：n&(~(1<<k));

成对变换：

注意到，当n为偶数时：n ^ 1 = n+1

当n为奇数时： n ^ 1=n-1

这里的技巧常用于无向图邻接表边集的存储中。因为相邻的两条边总是存储在临近的两个位置n和n+1上，而且n为偶数。那么n^1就可以得到与n边反向的边的编号了。

lowbit操作：

lowbit操作是取出数n在二进制表示下的最低位的1。假设n的第一个1在第k位上，操作流程如下：

首先，对n进行取反操作，这样0k-1位的所有0都变成了1，而第k位变成了0。接下来对取反后的数据+1，那么0k-1位的数就会进位到第k位，这里的0~k-1又变成0了，而第k位变成1了。此时k+1位到最高位的数与原n相反，按位与一下就可以全部清零了。

因此，lowbit(n)=n&(n+1) 而在补码表示法下，n=-1-n，因此有：

lowbit(n)=n & (-n);

这个操作是树状数组的基础。

hash操作存储对数结果：

定义hash数组h[x]，使得h[2^t]=t;相当于存储一个取对数表，可以在线性时间内完成。

如：

for(int i=1;i<=20;i++) h[(1<<i)]=i;

进一步，运用一个数学小技巧，即2^k mod 37对任意k∈[0,35]且k∈Z均不相等，可以这样存储：

for(int i=0;i<36;i++) h[(1<<i)%37]=i;
//相应的,访问时需要对数据取37的模
cout<<h[x%37]<<endl;

这个操作可以用来优化RMQ

前缀和与差分

对于数列{an}，它的前缀和{sn}，有sn=a1+a2+a3+…+an;称为一维前缀和，那么显然，前缀和可以方便地查询任意区间的总和。如:

sn=sn-1+an,sum(n,m)=sm-s(n-1);

进一步的，前缀和结论可以扩展到二维空间。设s[i] [j]表示矩阵(0,0)到(i,j)范围内所有元素的和，有结论：

Si,j=Si-1,j+Si,j-1-Si-1,j-1+ai,j

查询左下角(x,y)到右上角(m,n)内所有元素的和：

Sx,y->m,n=Sm,n-Sx-1,n-Sm,y-1+Sx-1,y-1

差分，可以理解为前缀和的逆运算，对差分数列求前缀和可以得到原数列中任意一项的值。对于差分数列

bn=an-an-1

b1+b2+..+bn=an

1.树状数组：

支持单点修改，区间查询，原理是用O(logn)的时间复杂度维护原序列的前缀和。

模板：https://www.luogu.com.cn/problem/P3374

#include<cstdio>
#define N 500010
using namespace std;
int c[N],n,m;
inline int lowbit(int x)
{
	return x&(-x);
}
inline void update(int x,int y)//对x位置上的数据加上y
{
	int i=x;
	while(i<=n)
	{
		c[i]+=y;
		i+=lowbit(i);
	}
} 
inline int query(int x)//询问1~x的前缀和 
{
	int i=x,sum=0;
	while(i)
	{
		sum+=c[i];
		i-=lowbit(i);
	}
	return sum;
}
int main()
{
	int t,x,y;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		update(i,x);
	}
	for(int i=1;i<=m;i++)
	{
		scanf("%d%d%d",&t,&x,&y);
		if(t==1)
		 update(x,y);
		if(t==2)
		 printf("%d\n",query(y)-query(x-1));
	}
	return 0;
}

扩展，对于二维的树状数组，可以加速维护二维的前缀和！写法就是在函数内再加一层循环即可

inline void update(int x,int y,int v)
{
    int i=x;
    while(i<=n)
    {
        int j=y;
        while(j<=n)
        {
            c[i][j]+=v;
            j+=lowbit(j);
        }
        i+=lowbit(i);
    }
}
inline int query(int x,int y)
{
    int sum=0,i=x;
    while(i)
    {
        int j=y;
        while(j)
        {
            sum+=c[i][j];
            j-=lowbit(j);
        }
        i-=lowbit(i);
    }
    return sum;
}

2.线段树：

支持区间修改，区间查询。复杂度为O(mlogn)

例题：https://www.luogu.com.cn/problem/P3373

维护区间加法、区间查询操作：

#include<cstdio>
#define Int register int
#define ll long long
using namespace std;
const int N=100020;
int n,m,a[N];
struct node{
	int add;
	ll sum;
}t[N<<2];
void build(int k,int l,int r)
{
	if(l==r)
		{
			t[k].add=0;
			t[k].sum=a[l];
			return;
		}
	int mid=l+r>>1;
	build(k<<1,l,mid);
	build(k<<1|1,mid+1,r);
	t[k].sum=t[k<<1].sum+t[k<<1|1].sum;
	return;
}
void add(int k,int l,int r,int val)//将k结点打上一个延迟标记val
{
	t[k].add+=val;
	t[k].sum+=(r-l+1)*val;//更新打标机之后对结点权值和的影响 
	return;
}
void pushdown(int k,int l,int r)//将k结点的标记下传到子树上,因为k结点的add标记之前作用与权值和了，
//不需要再进行重复计算。 
{
	if(t[k].add==0) return;//无标记就不需要进行标记下传
	int mid=l+r>>1;
	add(k<<1,l,mid,t[k].add);
	add(k<<1|1,mid+1,r,t[k].add);
	t[k].add=0;//标记清零 
}
void modify(int k,int l,int r,int x,int y,int val)
{
	if(x<=l && y>=r) return add(k,l,r,val);//如果当前区间被包括在修改区间内了，直接及逆行修改即可。 
	if(x>r || y<l) return;//在范围外部，回退 
	int mid=l+r>>1;
	pushdown(k,l,r);//否则，先进行标记下传 
	if(x<=mid) modify(k<<1,l,mid,x,y,val);//更新左右子树 
	if(y>mid) modify(k<<1|1,mid+1,r,x,y,val);
	t[k].sum=t[k<<1].sum+t[k<<1|1].sum; //更新sum 
}
ll query(int k,int l,int r,int x,int y)
{
	if(x<=l && r<=y) return t[k].sum;
	if(x> r || y<l) return 0;
	int mid=(l+r)>>1;
	pushdown(k,l,r);
	ll res=0;
	if(x<=mid) res+=query(k<<1,l,mid,x,y);
	if(y>mid) res+=query(k<<1|1,mid+1,r,x,y);
	return res; 
}
int main()
{
	int t,x,y,val;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	build(1,1,n);
	for(int i=1;i<=m;i++)
		{
			scanf("%d",&t);
			if(t==1)
				{
					scanf("%d%d%d",&x,&y,&val);
					modify(1,1,n,x,y,val);
					continue;
				}
			if(t==2)
				{
					scanf("%d%d",&x,&y);				
					printf("%lld\n",query(1,1,n,x,y));
					continue;
				}
		}
	return 0;
}

还有一种标记永久化的写法，在树套树和可持久化数据结构中比较常见：

#include<cstdio>
#define max(a,b) ((a)>(b)?(a):(b))
#define min(a,b) ((a)<(b)?(a):(b))
#define ll long long
using namespace std;
const int N=1e5+2;
struct node{
	ll add;
	ll sum;
}t[N<<2];
int n,m;
inline ll read()
{
	int f=1;
	char ch;
	while((ch=getchar())>'9' || ch<'0')
	{
		if(ch=='-') f=-1;
	}
	ll res=ch^48;
	while((ch=getchar())>='0'&&ch<='9')
		res=(res<<3)+(res<<1)+(ch^48);
	return res;
}
ll a[N];
int type;
void build(int k,int l,int r)
{
	if(l==r)
		{
			t[k].add=0;
			t[k].sum=a[l];
			return;
		}
	int mid=l+r>>1;
	build(k<<1,l,mid);
	build(k<<1|1,mid+1,r);
	t[k].sum=t[k<<1].sum+t[k<<1|1].sum;
	return;
}
void modify(int k,int l,int r,int x,int y,ll v)
{
	if(x>r || y<l) return;
	if(x<=l&&r<=y)
		{
			t[k].add+=v;
			return;
		}
	t[k].sum+=(min(y,r)-max(l,x)+1)*v;
	int mid=l+r>>1;
	if(x<=mid) modify(k<<1,l,mid,x,y,v);
	if(y>mid) modify(k<<1|1,mid+1,r,x,y,v);
	return;
}
ll query(int k,int l,int r,int x,int y)
{
	if(x>r || y<l) return 0;
	if(x<=l&&r<=y)
		return t[k].sum+(r-l+1)*t[k].add;
	ll res=t[k].add*(min(r,y)-max(l,x)+1);
	int mid=l+r>>1;
	if(x<=mid) res+=query(k<<1,l,mid,x,y);
	if(y>mid) res+=query(k<<1|1,mid+1,r,x,y);
	return res;
}
int main()
{
	n=read();
	m=read();
	for(int i=1;i<=n;i++)
		a[i]=read();
	build(1,1,n);
	ll x,y,val;
	for(int i=1;i<=m;i++)
		{
			scanf("%d",&type);
			if(type==1)
			{
				x=read();
				y=read();
				val=read();
				modify(1,1,n,x,y,val);
				continue;
			}
			if(type==2)
			{
				x=read();
				y=read();
				printf("%lld\n",query(1,1,n,x,y));
				continue;
			}
		}
	return 0;
}

还可以进一步，维护区间所有数同时乘上一个数，或者加上一个数，同时查询区间和。

https://www.luogu.com.cn/problem/P3373

#include <iostream>
#include <cstdio>
using namespace std;
//题目中给的p
int p;
//暂存数列的数组
long long a[100007];
//线段树结构体，v表示此时的答案，mul表示乘法意义上的lazytag，add是加法意义上的
struct node{
    long long v, mul, add;
}st[400007];
//buildtree
void bt(int root, int l, int r){
//初始化lazytag
    st[root].mul=1;
    st[root].add=0;
    if(l==r){
        st[root].v=a[l];
    }
    else{
        int m=(l+r)/2;
        bt(root*2, l, m);
        bt(root*2+1, m+1, r);
        st[root].v=st[root*2].v+st[root*2+1].v;
    }
    st[root].v%=p;
    return ;
}
//核心代码，维护lazytag
void pushdown(int root, int l, int r){
    int m=(l+r)/2;
//根据我们规定的优先度，儿子的值=此刻儿子的值*爸爸的乘法lazytag+儿子的区间长度*爸爸的加法lazytag
    st[root*2].v=(st[root*2].v*st[root].mul+st[root].add*(m-l+1))%p;
    st[root*2+1].v=(st[root*2+1].v*st[root].mul+st[root].add*(r-m))%p;
//很好维护的lazytag
    st[root*2].mul=(st[root*2].mul*st[root].mul)%p;
    st[root*2+1].mul=(st[root*2+1].mul*st[root].mul)%p;
    st[root*2].add=(st[root*2].add*st[root].mul+st[root].add)%p;
    st[root*2+1].add=(st[root*2+1].add*st[root].mul+st[root].add)%p;
//把父节点的值初始化
    st[root].mul=1;
    st[root].add=0;
    return ;
}
//update1，乘法，stdl此刻区间的左边，stdr此刻区间的右边，l给出的左边，r给出的右边
void ud1(int root, int stdl, int stdr, int l, int r, long long k){
//假如本区间和给出的区间没有交集
    if(r<stdl || stdr<l){
        return ;
    }
//假如给出的区间包含本区间
    if(l<=stdl && stdr<=r){
        st[root].v=(st[root].v*k)%p;
        st[root].mul=(st[root].mul*k)%p;
        st[root].add=(st[root].add*k)%p;
        return ;
    }
//假如给出的区间和本区间有交集，但是也有不交叉的部分
//先传递lazytag
    pushdown(root, stdl, stdr);
    int m=(stdl+stdr)/2;
    ud1(root*2, stdl, m, l, r, k);
    ud1(root*2+1, m+1, stdr, l, r, k);
    st[root].v=(st[root*2].v+st[root*2+1].v)%p;
    return ;
}
//update2，加法，和乘法同理
void ud2(int root, int stdl, int stdr, int l, int r, long long k){
    if(r<stdl || stdr<l){
        return ;
    }
    if(l<=stdl && stdr<=r){
        st[root].add=(st[root].add+k)%p;
        st[root].v=(st[root].v+k*(stdr-stdl+1))%p;
        return ;
    }
    pushdown(root, stdl, stdr);
    int m=(stdl+stdr)/2;
    ud2(root*2, stdl, m, l, r, k);
    ud2(root*2+1, m+1, stdr, l, r, k);
    st[root].v=(st[root*2].v+st[root*2+1].v)%p;
    return ;
}
//访问，和update一样
long long query(int root, int stdl, int stdr, int l, int r){
    if(r<stdl || stdr<l){
        return 0;
    }
    if(l<=stdl && stdr<=r){
        return st[root].v;
    }
    pushdown(root, stdl, stdr);
    int m=(stdl+stdr)/2;
    return (query(root*2, stdl, m, l, r)+query(root*2+1, m+1, stdr, l, r))%p;
}
int main(){
    int n, m;
    scanf("%d%d%d", &n, &m, &p);
    for(int i=1; i<=n; i++){
        scanf("%lld", &a[i]);
    }
    bt(1, 1, n);
    while(m--){
        int chk;
        scanf("%d", &chk);
        int x, y;
        long long k;
        if(chk==1){
            scanf("%d%d%lld", &x, &y, &k);
            ud1(1, 1, n, x, y, k);
        }
        else if(chk==2){
            scanf("%d%d%lld", &x, &y, &k);
            ud2(1, 1, n, x, y, k);
        }
        else{
            scanf("%d%d", &x, &y);
            printf("%lld\n", query(1, 1, n, x, y));
        }
    }
    return 0;
}

3.RMQ问题(Range Maximum/Minimum Question)：

对于给定静态序列，不进行后续操作，有多次询问区间[i,j]最大值/最小值(通常为10^6数量级)，可以使用ST算法进行O(nlogn)预处理，O(1)的询问。本质为动态规划。

预处理：

设f[i] [j]表示横坐标i为起点，向后走2^j的区间的最小值。显然这个区间可以用两个2^(j-1)得到，即状态转移方程为：

f[i] [j]=max(f[i] [j-1],f[i+(1<<j-1)] [j-1]);

设log[N]表示真数为N的log2对数值。可以有递推关系log[x]=log[x>>1]+1;注意赋值log[0]=-1,这样log[1]=0;

询问：

求出[X,Y]内的最大值：取k=log(Y-X+1),则ans=max(f[X] [k],f[Y-(1<<k)] [k]);

https://www.luogu.com.cn/problem/P3865

#include<cstdio>
#define max(a,b) ((a)>(b)?(a):(b))
using namespace std;
const int N=1e5+2,logN=20;
int n,m,f[N][logN],ans,log[N],X,Y;
inline int read()
{
	char ch;
	int f=1;
	while((ch=getchar())>'9' || ch<'0')
		{
			if(ch=='-') f=-1; 
		}
	int res=ch^48;
	while((ch=getchar())>='0'&&ch<='9')
		res=(res<<3)+(res<<1)+(ch^48);
	return res*f;
}
int main()
{
	n=read();
	m=read();
	log[0]=-1;
	for(int i=1;i<=n;i++)
	{
		f[i][0]=read();
		log[i]=log[i>>1]+1;
	}
	for(int j=1;j<=logN;j++)
		for(int i=1;i+(1<<j)-1<=n;i++)
				f[i][j]=max(f[i][j-1],f[i+(1<<j-1)][j-1]);
	for(int i=1;i<=m;i++)
		{
			X=read();
			Y=read();
			int k=log[Y-X+1];
			ans=max(f[X][k],f[Y-(1<<k)+1][k]);
			printf("%d\n",ans);
		}
	return 0;
}

4.倍增求LCA

树上倍增优化LCA求解速度。本质依旧是动态规划；

设f[i] [j]表示从i开始向上走2^j所到达的点的标号。显然有f[i] [j]=f[f[i] [j-1] ] [j-1];这里可以进行递推。而且f[i] [0]就是父结点。

其次，我们需要记录dep[x]也就是结点x的深度。赋值dep[root]=1;

算法步骤：

1.对于每个询问x,y，先深度大的x(否则，swap(x,y))向上跳，使得xy的深度一样

2.若x=y，直接return

3.x!=y,那么分别向上跳，从高幂次向低幂次跳跃，这样才能实现加速。当且仅当f[x] [t]!=f[y] [t]的时候才跳，这样最后f[x] [0]=f[y] [0]就是LCA了

#include<cstdio>
using namespace std;
const int N=500002,logN=20;
struct node{
	int from;
	int to;
	int next;
}e[N<<1];
int head[N<<1],num,n,m,s,dep[N],f[N][logN],log[N];
void addedge(int from,int to)
{
	e[++num].from=from;
	e[num].to=to;
	e[num].next=head[from];
	head[from]=num;
}
void dfs(int u,int fa)//预处理得到f数组和dep数组的值。 
{
	dep[u]=dep[fa]+1;
	f[u][0]=fa;
	for(int i=1;i<=log[dep[u]];i++)
		f[u][i]=f[f[u][i-1]][i-1];
	int v;
	for(int i=head[u];i;i=e[i].next)
	{
		v=e[i].to;
		if(v!=fa)
			dfs(v,u);
	}
}
void swap(int &a,int &b)
{
	int t;
	t=a;
	a=b;
	b=t;
	return;
}
int ask(int x,int y)
{
	if(dep[x]<dep[y]) swap(x,y);
	while(dep[x]>dep[y])
			x=f[x][log[dep[x]-dep[y]]];
	if(x==y) return x;
	for(int i=log[dep[x]];i>=0;i--)
	{
		if(f[x][i]!=f[y][i])
			{
				x=f[x][i];
				y=f[y][i];
			}
	}
	return f[x][0];
}
int main()
{
	int x,y;
	log[0]=-1;
	scanf("%d%d%d",&n,&m,&s);
	for(int i=1;i<=n-1;i++)
		{
			scanf("%d%d",&x,&y);
			addedge(x,y);
			addedge(y,x);
			log[i]=log[i>>1]+1;
		}
	log[n]=log[n>>1]+1;
	dfs(s,0);
	for(int i=1;i<=m;i++)
		{
			scanf("%d%d",&x,&y);
			printf("%d\n",ask(x,y));
		}
	return 0;
}

inline void build(int k,int l,int r)
{
	if(l==r)
		{
			sum[k]=Max[k]=num[rev[l]];
			return;
		}
	int mid=(l+r)>>1;
	build(k<<1,l,mid);
	build(k<<1|1,mid+1,r);
	sum[k]=sum[k<<1]+sum[k<<1|1];
	Max[k]=max(Max[k<<1],Max[k<<1|1]);
	return;
}
void update(int k,int l,int r,int pos,int val)
{
	if(pos<l||pos>r)
		return;
	if(l==r&&pos==l)
		{
			sum[k]=Max[k]=val;
			return;
		}
	int mid=(l+r)>>1;
	if(pos<=mid)
		update(k<<1,l,mid,pos,val);
	if(pos>mid)
		update(k<<1|1,mid+1,r,pos,val);
	sum[k]=sum[k<<1]+sum[k<<1|1];
	Max[k]=max(Max[k<<1],Max[k<<1|1]);
	return;
}
void query(int k,int l,int r,int L,int R)
{
	if(L>r || R<l)
		return;
	if(L<=l&&r<=R)
		{
			Summ+=sum[k];
			Maxx=max(Max[k],Maxx);
			return;
		}
	int mid=(l+r)>>1;
	if(L<=mid)
		query(k<<1,l,mid,L,R);
	if(R>mid)
		query(k<<1|1,mid+1,r,L,R);
	return ;
}
void ask(int x,int y)
{
	int fx=top[x],fy=top[y];
	while(fx!=fy)
		{
			if(dep[fx]<dep[fy])
				swap(x,y),swap(fx,fy);
			query(1,1,seg[0],seg[fx],seg[x]);
			x=fa[fx];
			fx=top[x];
		}
	if(dep[x]>dep[y])
		swap(x,y);
	query(1,1,seg[0],seg[x],seg[y]);
}

7.Treap平衡树

Treap平衡树的本质是一个二叉查找树Tree的基础上，对每个节点都赋予一个随机权值，再使得每个节点保证构成小根堆Heap，从而避免了有序数据使得树退化为链的情况。Tree+Heap–>Treap

操作一：左旋右旋操作：Zag/Zig

为了在维持BST性质的前提下，还要构成小根堆的性质，我们势必要对结点进行操作。这里的操作就是旋转。

Zig操作，右旋：对于以k为根节点的子树，将k的左子结点变为新的根节点，原左子结点的右子结点变成k的新左子结点，而新根的新右子结点变为k。右旋的形象化表现就是k的左子结点被抬升了上去。

Zag操作，左旋：对于以k为根节点的子树，将k的右子结点变为新的根节点，原右子结点的左子结点变为k的新右子结点，而新根的新左子结点变为k。左旋的形象化表现就是k的右子结点被抬升了上去。

代码如下：

	#define cnt(x) t[x].cnt

操作二：将权值为w的结点插入到Treap中

算法如下：

现需要找到权值为w结点插入的位置，从根节点开始遍历，将w与当前结点的权值进行比较：若w==val(x) size(x)++,cnt(x)++;

如果w<val(x) 那么就插入到左子树中；如果w>val(x)那么就插入到右子树中；

如果到达一个空的结点，那就新建一个结点。使得其值为w，并且随机一个优先值p(x)=rand();

在回溯的过程中进行左右旋的操作：

inline void insert(int &k,const int &key)
{
	if(!k)
		{
			k=++pool;
			val(k)=key;
			cnt(k)=size(k)=1;
			p(k)=rand(); 
			lc(k)=rc(k)=0;
			return;
		}
	else ++size(k);
	
	if(key==val(k))
	{
			cnt(k)++;
	}
	if(key<val(k))
	{
		insert(lc(k),key);
		if(p(lc(k))<p(k))
			Zig(k);
	}
	if(key>val(k))
	{
		insert(rc(k),key);
		if(p(rc(k))<p(k))
			Zag(k);
	}
	return;
}

可以看出，左右旋操作在维护小根堆性质中的重要作用！

操作三：删除操作

算法流程：

先找到需要删除元素在Treap中的位置。

其次，若cnt(k)>1 直接cnt(k)–,size(k)–

若cnt(k)==1，分两种情况分析：

若该节点为链结点或叶节点，那么就直接将其删除即可。k=lc(k)+rc(k)

若该节点有两个非空左右子结点，我们的思路就是，比较左右两个子结点的优先值的大小，将优先值较小的结点旋转到该节点处，再对旋转后的子树进行这样的递归操作，直到该节点变为叶节点或者链结点再执行方案一

inline void Delete(int &k,const int &key)
{
	size(k)--;
	if(key<val(k)) Delete(lc(k),key);
	if(key>val(k)) Delete(rc(k),key);
	if(key==val(k))
	{
		if(cnt(k)>1){
			cnt(k)--;
			return;
		}
		if(!lc(k) || !rc(k))//如果k结点是链节点或者叶节点
			{
				k=lc(k)+rc(k);
			}
		else
			{
				
				if(p(lc(k))<p(rc(k)))//如果左儿子更优，那就右旋一下
					Zig(k),Delete(k,key); 
				else
					Zag(k),Delete(k,key);
			} 
	}
	return;
}

操作四：求一个元素在Treap中的前驱，也就是最大的小于等于该元素的Treap中的数据。

只需要不断对该元素w与结点值进行比较就可以了，从根结点出发，遍历结点即可。

inline int QueryPre(const int &key)
{

		else x=lc(x);
	}

操作五：查找Treap中排名第k大的元素值：

我们思考第k大元素的位置。由于整颗Treap是有序的，若存在结点x，使得size(lc(x))+1=k,那么结点x就是对应的第k大元素。这里对于相同元素我们默认其排序是相同的。那么，对于val(k)的结点，占据的区间便是：[size(lc(x))+1,size(lc(x))+cnt(x)];

于是，我们可以按照分治的思路寻找第k大的元素：

若k∈[size(lc(x))+1,size(lc(x))+cnt(x)],返回val(x)

若k<=size(lc(x)) 那么就在左子树中寻找对应元素，且排序依旧是第k个元素

若k>size(lc(x))+cnt(x)，那么就在右子树中寻找对应元素，且排序是第k-size(lc(x))-cnt(x)个元素

inline int QueryKth(int k)
{
	int x=root;
	while(x)
	{
		if(k>=size(lc(x))+1&&k<=size(lc(x))+cnt(x)) return val(x); 
		if(k<=size(lc(x))) 
		{
			x=lc(x);
		}
		else 
		{
			k=k-size(lc(x))-cnt(x);
			x=rc(x);
		}
	}
	return 0;
}

操作六：查找Treap中元素值为w的元素的排名；

可以看作操作五的逆运算过程！一样用分治的思路：res代表已经累计的小于w的元素个数，只需要考虑当前以k为根的子树即可。

若w=val(k) return res+size(lc(k))+1;

若w<val(k) 就跳到左子树中去寻找。 k=lc(k);

若w>val(k) 就跳到右子树中去寻找。同时res需要加上所有左子树和根节点的大小！res+=size(lc(k))+cnt(k);k=rc(k);

inline int QueryRank(int w)
{
	int x=root,res=0;
	while(x)
	{
		if(w==val(x)) return res+size(lc(x))+1;
		if(w<val(x)) x=lc(x);
		else res+=size(lc(x))+cnt(x),x=rc(x);
	}
	return res;
}
